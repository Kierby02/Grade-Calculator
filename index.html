from pyscript import Element def calculate_grades(event): try: attendance = float(Element("attendance").element.value) * 0.1 recitation = float(Element("recitation").element.value) * 0.4 quiz = float(Element("quiz").element.value) * 0.3 requirements = float(Element("requirements").element.value) * 0.3 prelim_exam = float(Element("prelim_exam").element.value) * 0.6 performance_task = recitation + quiz + requirements overall_performance = performance_task * 0.3 prelim_grade = round(attendance + overall_performance + prelim_exam, 2) Element("result").write(f"Your Prelim Grade is: {prelim_grade:.2f}") Element("result").element.style.display = "block" except ValueError: Element("result").write("Invalid input. Please enter numbers only.") Element("result").element.style.display = "block" def populate_inputs(event): Element("attendance").element.value = 90 Element("recitation").element.value = 100 Element("quiz").element.value = 95 Element("requirements").element.value = 90 Element("prelim_exam").element.value = 80 def reset_inputs(event): for element_id in ["attendance", "recitation", "quiz", "requirements", "prelim_exam"]: Element(element_id).element.value = '' Element("result").element.style.display = "none" Element("calculateButton").element.onclick = calculate_grades Element("populateButton").element.onclick = populate_inputs
